## 4. API Response Envelope

Every API response uses this shape. No exceptions.

```js
// utils/apiResponse.js

export const success = (res, data, message = 'Success', statusCode = 200) =>
  res.status(statusCode).json({ success: true, message, data, errors: null });

export const paginated = (res, data, pagination, message = 'Success') =>
  res.status(200).json({
    success: true, message, data, errors: null,
    pagination: {
      total: pagination.total,
      page: pagination.page,
      limit: pagination.limit,
      totalPages: Math.ceil(pagination.total / pagination.limit),
    },
  });

export const error = (res, message = 'An error occurred', statusCode = 500, errors = null) =>
  res.status(statusCode).json({ success: false, message, data: null, errors });
```

### Response examples

**Single resource — 200**
```json
{
  "success": true,
  "message": "Product created successfully",
  "data": { "id": 1, "name": "Engine Oil 5W-30", "sku": "EO5W30", "costPrice": 2340, "salePrice": 2800 },
  "errors": null
}
```

**Paginated list — 200**
```json
{
  "success": true,
  "message": "Products fetched successfully",
  "data": [ ],
  "errors": null,
  "pagination": { "total": 84, "page": 1, "limit": 20, "totalPages": 5 }
}
```

**Validation error — 422**
```json
{
  "success": false,
  "message": "Validation failed",
  "data": null,
  "errors": [
    { "field": "salePrice", "message": "Sale price must be greater than 0" }
  ]
}
```

**Forbidden — 403**
```json
{ "success": false, "message": "Forbidden — insufficient role", "data": null, "errors": null }
```

**Not found — 404**
```json
{ "success": false, "message": "Product not found", "data": null, "errors": null }
```

**Server error — 500**
```json
{ "success": false, "message": "An unexpected error occurred", "data": null, "errors": null }
```

## 7. Auth Strategy

- **Access Token:** JWT, expires `15m`. Header: `Authorization: Bearer <token>`.
- **Refresh Token:** JWT, expires `7d`. HTTP-only cookie.
- **Role hierarchy:** `ADMIN > MANAGER > CASHIER > VIEWER`.
- **JWT is parsed only inside `auth.middleware.js`.** Never in controllers or services.

```js
// middleware/auth.middleware.js
import jwt from 'jsonwebtoken';
import { ROLES } from '../config/constants.js';
import { error } from '../utils/apiResponse.js';

export const verifyToken = (req, res, next) => {
  const token = req.headers.authorization?.split(' ')[1];
  if (!token) return error(res, 'No token provided', 401);
  try {
    req.user = jwt.verify(token, process.env.JWT_SECRET);
    next();
  } catch {
    return error(res, 'Invalid or expired token', 401);
  }
};

// requireRole('MANAGER') allows MANAGER and ADMIN
export const requireRole = (...roles) => (req, res, next) => {
  const hierarchy = [ROLES.VIEWER, ROLES.CASHIER, ROLES.MANAGER, ROLES.ADMIN];
  const minLevel = Math.min(...roles.map(r => hierarchy.indexOf(r)));
  if (hierarchy.indexOf(req.user.role) < minLevel)
    return error(res, 'Forbidden — insufficient role', 403);
  next();
};
```

## 10. Business Logic Rules

### 10.1 Sale Creation — TRANSACTION REQUIRED

```
Uses prisma.$transaction(). All steps succeed or all fail.

1.  Validate all items: product exists, isActive = true, isDeleted = false.
2.  If paymentMethod is CREDIT or SEMI_CREDIT: customerId is required.
3.  For each item (read product once, use values throughout):
    a. Fetch the full product record from DB.
    b. If cartUnitPrice provided:
         unitPrice = cartUnitPrice
         originalUnitPrice = product.salePrice
         isPriceOverridden = true
         priceOverriddenById = req.user.id
       Else:
         unitPrice = product.salePrice
         originalUnitPrice = product.salePrice
         isPriceOverridden = false
    c. costPrice         = product.costPrice   ← FROZEN now
    d. productName       = product.name        ← FROZEN now
    e. sku               = product.sku         ← FROZEN now
    f. totalPrice        = (unitPrice - discount) × quantity
    g. profitAmount      = (unitPrice - costPrice - discount) × quantity
4.  subTotal       = Σ item.totalPrice
5.  totalAmount    = subTotal - orderDiscountAmount + taxAmount
6.  totalCost      = Σ (item.costPrice × item.quantity)
7.  totalProfit    = Σ item.profitAmount - orderDiscountAmount
8.  Derive payment fields:
      CASH:        changeGiven = amountTendered - totalAmount; amountOnCredit = 0
      CARD:        amountTendered = totalAmount; changeGiven = 0; amountOnCredit = 0
      CREDIT:      amountTendered = 0; changeGiven = 0; amountOnCredit = totalAmount
      SEMI_CREDIT: amountOnCredit = totalAmount - amountTendered; changeGiven = 0
9.  invoiceNumber = await generateInvoiceNumber(tx)
10. Create SaleOrder + SaleOrderItems (inside tx).
11. For each item where product.isService === false:
      inventory.service.js: createStockMovement(tx, MOVEMENT_TYPES.SALE, ...)
      inventory.service.js: decrementStock(tx, productId, quantity)
12. If customerId: tx.customer.update({ where: { id: customerId },
      data: { totalSpent: { increment: totalAmount } } })
13. If amountOnCredit > 0: tx.customer.update({ where: { id: customerId },
      data: { creditBalance: { increment: amountOnCredit } } })
```

### 10.2 Price Change — TRANSACTION REQUIRED

```
Uses prisma.$transaction(). Both steps succeed or both fail.

1. Fetch current product.
2. newCostPrice = calculateCostPrice(newPurchasePrice, newPurchaseTaxPercent)
   — calculateCostPrice lives in pricing.service.js. Import from there.
3. Create ProductPriceHistory row (all old values, all new values, changedById, reason).
4. Update Product (purchasePrice, purchaseTaxPercent, costPrice, salePrice, updatedById).
```

### 10.3 GRN (Goods Received) — TRANSACTION REQUIRED

```
Uses prisma.$transaction().

1. PO status must be SENT or PARTIALLY_RECEIVED.
2. For each item in the receive request:
   a. receivedQty ≤ (orderedQty - item.receivedQty already on record).
   b. inventory.service.js: createStockMovement(tx, MOVEMENT_TYPES.PURCHASE, ...)
   c. inventory.service.js: incrementStock(tx, productId, receivedQty)
   d. tx.purchaseOrderItem.update: item.receivedQty += incomingQty  // incomingQty = qty from request for this item
3. Recalculate status:
     All items fully received → RECEIVED, set receivedAt = now
     Else → PARTIALLY_RECEIVED
```

### 10.4 Sale Void — TRANSACTION REQUIRED

```
Uses prisma.$transaction().
Role: MANAGER or ADMIN only.

1. voidReason required.
2. Sale status must be COMPLETED. Cannot void an already-voided sale.
3. Set status = VOIDED, voidedAt = now, voidedById = req.user.id, voidReason.
4. For each item where product.isService === false:
   inventory.service.js: createStockMovement(tx, MOVEMENT_TYPES.SALE_VOID, ...)
   inventory.service.js: incrementStock(tx, productId, item.quantity)
5. If sale.customerId AND sale.amountOnCredit > 0:
   tx.customer.update: creditBalance -= sale.amountOnCredit
6. If sale.customerId:
   tx.customer.update: totalSpent -= sale.totalAmount
   // Walk-in sales (no customerId) skip steps 5 and 6 entirely
```

### 10.5 End of Day

```
No transaction needed (aggregate read, one write at end).
Role: MANAGER or ADMIN. Passcode verified by frontend before API call.

1. Guard: check EndOfDaySummary for today's date. Error if exists.
2. Query all SaleOrders: createdAt >= todayMidnight AND createdAt <= now.
3. Aggregate across completed orders only (status = COMPLETED):
   - Count total vs voided
   - Sum totalAmount by paymentMethod → cashRevenue, creditRevenue, etc.
   - Sum totalProfit
   - Sum amountTendered for CASH and SEMI_CREDIT orders → cashSales
   - Sum changeGiven
   - Top 5 products by quantity sold (use SaleOrderItems)
   - Top 5 products by revenue (use SaleOrderItems)
4. expectedClosingCash = openingCash (from req.body) + cashSales - changeGiven
5. actualClosingCash from req.body (physically counted by manager)
6. cashDiscrepancy = actualClosingCash - expectedClosingCash
7. newCreditIssued = sum of amountOnCredit across completed CREDIT + SEMI_CREDIT orders
8. Create and return EndOfDaySummary.
```

## 11. Transaction Requirements

**These operations MUST use `prisma.$transaction()`. No exceptions.**

| Operation | Reason |
|-----------|--------|
| Sale creation | SaleOrder + SaleOrderItems + StockMovements + Inventory + Customer updates — all or nothing |
| Sale void | Stock restoration + Customer balance reversal must be atomic |
| Price change | ProductPriceHistory insert + Product update must be atomic |
| GRN receiving | StockMovements + Inventory + PO item updates must be atomic |

**Standard transaction pattern:**

```js
// In a service function
import prisma from '../config/db.js';

export const createSale = async (saleData, performedBy) => {
  return prisma.$transaction(async (tx) => {
    // Use tx for ALL Prisma calls inside this function
    const sale = await tx.saleOrder.create({ data: { ... } });

    // Pass tx to other service functions that participate in this transaction
    await decrementStock(tx, productId, quantity);

    return sale;
  });
};

// Service functions that can participate in transactions accept tx as first param
export const decrementStock = async (tx, productId, quantity) => {
  // tx is used here, not the global prisma instance
  const inventory = await tx.inventory.update({ ... });
  await tx.stockMovement.create({ ... });
  return inventory;
};
```

**Note on local PostgreSQL:** Transactions work out of the box on a local PostgreSQL installation. No replica set or special configuration required. This is a key advantage over MongoDB.

## 12. Invoice & PO Number Generation

Uses the `Counter` table with a raw SQL atomic increment to prevent race conditions.

```js
// utils/generateInvoiceNumber.js
import { format } from 'date-fns';

// tx can be a transaction client or the global prisma instance
// NOTE: $queryRaw returns BigInt for PostgreSQL integers — cast with Number()
export const generateInvoiceNumber = async (tx) => {
  const result = await tx.$queryRaw`
    UPDATE "Counter" SET value = value + 1 WHERE name = 'invoice' RETURNING value
  `;
  const seq = Number(result[0].value);  // Cast BigInt → Number
  return `INV-${format(new Date(), 'yyyyMMdd')}-${String(seq).padStart(4, '0')}`;
};

// utils/generatePONumber.js
import { format } from 'date-fns';
export const generatePONumber = async (tx) => {
  const result = await tx.$queryRaw`
    UPDATE "Counter" SET value = value + 1 WHERE name = 'po' RETURNING value
  `;
  const seq = Number(result[0].value);  // Cast BigInt → Number
  return `PO-${format(new Date(), 'yyyyMMdd')}-${String(seq).padStart(4, '0')}`;
};
```

**Seed the Counter table in Cycle 1:**
```js
await prisma.counter.upsert({ where: { name: 'invoice' }, create: { name: 'invoice', value: 0 }, update: {} });
await prisma.counter.upsert({ where: { name: 'po' },      create: { name: 'po', value: 0 },      update: {} });
```

## 13. Passcode System

PIN protection layered on top of role auth for sensitive UI modules.

### Flow
1. Admin sets a 4–6 digit PIN per module via `PUT /api/settings/passcode`. Stored as bcrypt hash in `Passcode` table.
2. Navigating to a protected module shows `<PasscodeModal module="..." onSuccess={...} />`.
3. Frontend calls `POST /api/settings/passcode/verify` with `{ module, pin }`.
4. Backend runs `bcrypt.compare(pin, storedHash)` → returns `{ success: true/false }`.
5. On success, access granted **in memory only** — expires on page refresh.

### Protected modules
| Module | What it gates |
|--------|--------------|
| `PROFIT_VIEW` | ProfitReport page, profit figures in EOD view |
| `STOCK_ENTRY` | Creating new products with opening stock |
| `STOCK_UPDATE` | Manual stock adjustment modal |
| `END_OF_DAY` | Entire EndOfDayScreen |
| `PRICE_OVERRIDE` | Cart item price editing (admin-toggleable) |

### API
```
POST   /api/settings/passcode/verify   { module, pin }   → { success: bool }
PUT    /api/settings/passcode          { module, pin }   → set PIN     [ADMIN]
DELETE /api/settings/passcode/:module                    → disable     [ADMIN]
```
